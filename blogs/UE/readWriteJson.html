官方文档

读 ​

json最外层结构是array 则用该方法

js
TArray<TSharedPtr<FJsonValue>>  jsonArray;

json最外层结构是object,则用该方法

js
TSharedPtr<FJsonObject> jsonObject

FJsonSerializer::Deserialize 判断是否读取成功

js
TSharedRef<TJsonReader<TCHAR>> JsonReader = TJsonReaderFactory<TCHAR>::Create(JsonString);
FJsonSerializer::Deserialize(JsonReader, jsonObject) // obj
FJsonSerializer::Deserialize(JsonReader, jsonArray)  // arr

获取json对象中不同类型的字段

js
// 获取object
TSharedPtr<FJsonObject> obj = jsonObject->GetObjectField("obj");
// 获取array
TArray<TSharedPtr<FJsonValue>> arr = jsonObject->GetArrayField("arr");
// 获取string
jsonObject->GetStringField("str");
// 获取Number
jsonObject->GetNumberField("num");

获取Array对应中不同类型的值

js
// 获取obj
jsonArray->AsObject()
// 获取arr
jsonArray->AsArray()
// 获取bool
jsonArray->AsBool()
// 获取num
jsonArray->AsNumber()
// 获取str
jsonArray->AsString()
