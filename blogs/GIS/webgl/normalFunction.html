镜像(Three) ​

js
 const mesh2 = mesh.clone();
      // mesh2.position.x = -mesh.position.x;
      mesh2.name = mesh.name.replace("left", "right");
      var m = new THREE.Matrix4();
      var vec = new THREE.Vector3(1, 0, 0);

      m.set(
        1 - 2 * vec.x * vec.x,
        -2 * vec.x * vec.y,
        -2 * vec.x * vec.z,
        0,
        -2 * vec.x * vec.y,
        1 - 2 * vec.y * vec.y,
        -2 * vec.y * vec.z,
        0,
        -2 * vec.x * vec.z,
        -2 * vec.y * vec.z,
        1 - 2 * vec.z * vec.z,
        0,
        0,
        0,
        0,
        1
      );
      mesh2.applyMatrix(m);
      mesh2.translateX(0);
      window.scene.add(mesh2);
